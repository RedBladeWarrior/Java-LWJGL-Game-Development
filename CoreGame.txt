package com.youtube.game;

public class CoreGame
{
	public final static int TILE_SIZE = 64;
	
	private static Player player;
	
	public CoreGame()
	{
		new Image();
		player = new Player();
	}
	
	public void input()
	{
		player.input();
	}
	
	public void logic()
	{
		player.logic();
	}
	
	public void render()
	{
		player.render();
	}
	
	public void dispose()
	{
		player.dispose();
		Image.dispose();
	}
}
